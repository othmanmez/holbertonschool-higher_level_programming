import unittest
from models.base_geometry import BaseGeometry  # Assurez-vous que le chemin est correct

class TestBaseGeometry(unittest.TestCase):
    def setUp(self):
        self.bg = BaseGeometry()

    def test_area(self):
        with self.assertRaises(Exception) as e:
            self.bg.area()
        self.assertEqual(str(e.exception), "area() is not implemented")

    def test_integer_validator_valid(self):
        self.bg.integer_validator("width", 10)  # Ne doit pas lever d'exception

    def test_integer_validator_not_integer(self):
        with self.assertRaises(TypeError) as e:
            self.bg.integer_validator("size", "5")
        self.assertEqual(str(e.exception), "size must be an integer")

    def test_integer_validator_negative(self):
        with self.assertRaises(ValueError) as e:
            self.bg.integer_validator("height", -5)
        self.assertEqual(str(e.exception), "height must be greater than 0")

if __name__ == "__main__":
    unittest.main()
